#include <iostream>//подключаем библиотеки и так далее
#include <vector>
#include <Windows.h>
using namespace std;
int main() {
	setlocale(LC_ALL, "Russian");//для того чтобы я мог вывести по русски
	cout << "Vvedite kol elementov massiva  ";
	int n; 
	cin >> n;
	vector <int> a(n);// O(n)
	vector <int> s; // O(1)
	cout << "Vvedite massiv  ";
	for (int i = 0; i < n; i++) {// O(n)
		cin >> a[i];
	}
	for (int i = 0; i < n; i++) {// O(n^2)
		bool flag = true;
		for (int j = i + 1; j < n; j++) {
			if (a[i] == a[j]) {
				flag = false;
				break;
			}
		}
		if (flag == true) {
			s.push_back(a[i]); // O(1) при равномерном распределении
		}
	}
	for (int i = 0; i < s.size(); i++) { // O(d) где d - количесто уникальных значений в s
		cout << s[i];
	}
}
// сложность алгоритма O(n) + O(n^2) + O(d) = O(n^2)
